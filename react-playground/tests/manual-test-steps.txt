TEST 2: Carousel with Framer Motion
========================================
1. Clear the editor
2. Paste this code:

import React, { useState } from 'react';
import { motion } from 'framer-motion';
import { FiChevronLeft, FiChevronRight } from 'lucide-react';

const Carousel = () => {
  const [activeIndex, setActiveIndex] = useState(0);
  const colors = ['bg-blue-500', 'bg-red-500', 'bg-green-500'];

  return (
    <div className="w-full h-screen flex overflow-hidden relative">
      {colors.map((color, index) => (
        <motion.div
          key={index}
          initial={{ x: index === activeIndex ? 0 : '100%' }}
          animate={{ x: index === activeIndex ? 0 : '100%' }}
          transition={{ duration: 0.5 }}
          className={`absolute top-0 left-0 w-full h-full ${color} flex items-center justify-center text-white text-6xl font-bold`}
          style={{ display: index === activeIndex ? 'flex' : 'none' }}
        >
          Slide {index + 1}
        </motion.div>
      ))}
      <div className="absolute top-0 left-0 w-full h-full flex justify-between items-center px-8">
        <button
          className="bg-white/80 p-4 rounded-full hover:bg-white"
          onClick={() => setActiveIndex((prev) => (prev - 1 + colors.length) % colors.length)}
        >
          <FiChevronLeft size={32} />
        </button>
        <button
          className="bg-white/80 p-4 rounded-full hover:bg-white"
          onClick={() => setActiveIndex((prev) => (prev + 1) % colors.length)}
        >
          <FiChevronRight size={32} />
        </button>
      </div>
    </div>
  );
};

export default Carousel;

3. Click "Run Code"
4. EXPECTED: Carousel renders with colored slides
5. EXPECTED: Console shows "SUCCESS"
6. EXPECTED: No errors in browser console
7. EXPECTED: Browser console shows "Loaded imports: {FiChevronLeft, FiChevronRight}"
8. Test navigation buttons work (left/right arrows)
